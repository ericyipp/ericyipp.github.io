import pandas as pd
import numpy as np

# Load the data
df = pd.read_csv('lung_cancer_survey.csv')

# Convert LUNG_CANCER to numerical: NO = 0, YES = 1
df['LUNG_CANCER_NUM'] = (df['LUNG_CANCER'] == 'YES').astype(int)

# Calculate correlation
smoking_correlation = np.corrcoef(df['SMOKING'], df['LUNG_CANCER_NUM'])[0, 1]
print(f"Correlation between SMOKING and LUNG_CANCER: {smoking_correlation}")

Correlation between SMOKING and LUNG_CANCER: 0.05817888585203873

smokers_with_cancer = ((df['SMOKING'] == 2) & (df['LUNG_CANCER'] == 'YES')).sum()

total_smokers = (df['SMOKING'] == 2).sum()
total_nonsmokers = (df['SMOKING'] == 1).sum()

print(f"Percentage of smokers with lung cancer: {smokers_with_cancer / total_smokers * 100:.2f}%")
print(f"Percentage of non-smokers with lung cancer: {nonsmokers_with_cancer / total_nonsmokers * 100:.2f}%")
